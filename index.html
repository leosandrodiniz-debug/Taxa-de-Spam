function renderCharts() { const projId = DADOS.activeProjectId; // 1. Filtra APENAS as datas que possuem histórico real para o projeto selecionado let datasDoProjeto = []; Object.keys(LOGS_PLANILHA).forEach(k => { let temLogDoProjeto = LOGS_PLANILHA[k].some(l => extractProjectId(l.id) === projId); if (temLogDoProjeto) { datasDoProjeto.push(k); } }); // 2. Ordena cronologicamente essas datas específicas datasDoProjeto.sort((a,b) => { let ptsA = a.split("/"); let ptsB = b.split("/"); return new Date(2026, parseInt(ptsA[1])-1, parseInt(ptsA[0])) - new Date(2026, parseInt(ptsB[1])-1, parseInt(ptsB[0])); }); // 3. Limita a exibição aos últimos 15 dias de atividade real para não espremer a tela let timelineEixoSpam = datasDoProjeto.slice(-15); if (timelineEixoSpam.length === 0) { if (activeCharts.gSpamDia) activeCharts.gSpamDia.destroy(); return; } // 4. Calcula a média real de spam por dia para este projeto let valoresEixoSpam = []; timelineEixoSpam.forEach(k => { let logs = LOGS_PLANILHA[k] || []; let projectLogs = logs.filter(l => extractProjectId(l.id) === projId); let sum = 0; projectLogs.forEach(l => sum += l.spam); valoresEixoSpam.push(parseFloat((sum / projectLogs.length).toFixed(3))); }); if (activeCharts.gSpamDia) activeCharts.gSpamDia.destroy(); try { activeCharts.gSpamDia = new Chart(document.getElementById('gSpamDia'), { type: 'line', data: { labels: timelineEixoSpam, datasets: [ { label: 'Spam % Médio Diário', data: valoresEixoSpam, borderColor: '#028090', backgroundColor: 'rgba(2, 128, 144, 0.08)', pointRadius: 5, pointHoverRadius: 7, tension: 0.2, fill: true }, { label: 'Teto Crítico (0.30%)', data: Array(timelineEixoSpam.length).fill(0.30), borderColor: '#E63946', borderDash: [5, 5], pointRadius: 0, fill: false } ] }, options: { responsive: true, maintainAspectRatio: false, scales: { y: { beginAtZero: true, // Força o gráfico a iniciar em 0% de spam suggestedMax: 0.40, // Mantém uma folga visual acima do teto de 0.30% ticks: { callback: function(value) { return value.toFixed(2) + '%'; // Coloca a % no eixo Y } } } }, plugins: { tooltip: { callbacks: { label: function(context) { return context.dataset.label + ': ' + context.raw.toFixed(3) + '%'; } } } } } }); } catch(e) { console.error("Erro ao gerar gráfico:", e); } }
